# ---------- BUILD STAGE ----------
FROM mcr.microsoft.com/dotnet/sdk:9.0 AS build
WORKDIR /src

COPY ./CorporateTEMPLATE/*.sln ./
COPY ./CorporateTEMPLATE/ ./CorporateTEMPLATE/

WORKDIR /src/CorporateTEMPLATE

RUN dotnet restore
RUN dotnet publish -c Release -o /app/publish

# ---------- RUNTIME STAGE ----------
FROM mcr.microsoft.com/dotnet/aspnet:9.0 AS final
WORKDIR /app

COPY --from=build /app/publish .

ENV ASPNETCORE_URLS=http://0.0.0.0:8080
EXPOSE 8080

ENTRYPOINT ["dotnet", "CorporateTEMPLATE.dll"]
